Example 1: Creative Writing

response = litellm.completion(
model="gpt-3.5-turbo",
messages=[{"role": "user", "content": "Write a haiku about programming"}],
max_tokens=60,
temperature=0.7
)

response = litellm.completion(
model="gpt-3.5-turbo",
