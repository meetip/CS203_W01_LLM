Python Code Template

import litellm

def rewrite_for_audience(text, target_style):
    # The system message guides the model's tone
    response = litellm.completion(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system",
             "content":
             f"Rewrite text in {target_style} 
              style while preserving key information."},
            {"role": "user",
             "content": f"Rewrite: {text}"}
        ],
        max_tokens=150,
        temperature=0.4
    )
    return response.choices[0].message.content
